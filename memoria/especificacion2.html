Recorrido con bucles


from tyger import TensorNp
import numpy as np
import torch

# NumPy
tensor: TensorNp[2, 2] = np.array([[1.0, 2.0], [3.0, 4.0]])
type(tensor) # TensorNp

first: Iterable = tensor[0] # [1.0, 2.0]
print(len(first)) # 2
second: Iterable = tensor[1] # [3.0, 4.0]

# [first, second] # Matriz representada en Iterables

# Recorremos la matriz y mostramos sus resultados
for t in tensor:
    type(t) # Iterable
    print(t) #
> [1.0, 2.0]
> [3.0, 4.0]

# Recorremos la matriz y mostramos sus resultados a traves de sus indices
for i in len(first):
    for j in len(second):
        print(tensor[i][j])

> 1.0
> 2.0
> 3.0
> 4.0

from tyger import TensorNp
import numpy as np
import torch

# NumPy
tensor: TensorNp[2, 2] = np.array([[1.0, 2.0], [3.0, 4.0]])
